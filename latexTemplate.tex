\begin{frame}{Theorems \& proofs}
  
  \begin{theorem}
    $e^{i\pi}+1=0$ \label{euler}
  \end{theorem}

  The proof of Theorem \ref{euler} is as below:
  
  \begin{proof}
    \begin{align*}
      e^{i\pi}+1&=\cos\pi+i\sin\pi+1\\
                &=-1+i\times0+1=0
    \end{align*}
  \end{proof}

\end{frame}

\begin{frame}{Using references}

  \begin{itemize}
  \item We can cite others via bibliography management.
  \item This means putting the references in a \texttt{.bib} file, and using a \texttt{\textbackslash{}cite$^{*}$} command when citing.
  \item Example: \citet{dbc21}
  \end{itemize}
